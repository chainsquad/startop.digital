Wir lieben #self-hosted

Es gibt ein #docker image? Wir liefern IP-Adresse und Port - natürlich mit sicherer #ssl Verschlüsselung. 

version: 3
services:
  database:
    image: postgres
  app:
    build: app/
    ports:
      - 8080:8080
  loadbalancer:
    image: nginx
      - 80:80
      - 443:443
